CREATE TABLE test.user
(
id	int not null AUTO_INCREMENT,
username	Varchar(20)	,
password	Varchar(20)	,
email	Varchar(50)	,
PRIMARY KEY (Id)
);

CREATE TABLE test.notes
(
id	int not null AUTO_INCREMENT,
user_Id int,
modify_time TIMESTAMP DEFAULT '0000-00-00 00:00:00' NOT NULL,
content	Varchar(1000)	,
status	Varchar(1) ,
create_time TIMESTAMP DEFAULT 'CURRENT_TIMESTAMP' NOT NULL,
reminder_time TIMESTAMP DEFAULT '0000-00-00 00:00:00' NOT NULL,
PRIMARY KEY (Id)
);

CREATE TABLE test.notes_history
(
id	int not null AUTO_INCREMENT,
notes_id int ,
modify_time TIMESTAMP DEFAULT '0000-00-00 00:00:00' NOT NULL,
content	Varchar(1000) ,
status	Varchar(1) ,
reminder_time TIMESTAMP DEFAULT 'CURRENT_TIMESTAMP' NOT NULL,
PRIMARY KEY (Id)
);
